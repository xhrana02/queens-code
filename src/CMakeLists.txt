file(GLOB_RECURSE _SRC_FILES "*.h" "*.cpp")
file(GLOB_RECURSE _ALL_FILES LIST_DIRECTORIES true "*")

set(_INCLUDE_DIRS "")
FOREACH(_FILE ${_ALL_FILES})
  IF(IS_DIRECTORY ${_FILE})
    list(APPEND _INCLUDE_DIRS ${_FILE})
  ENDIF()
ENDFOREACH()

set(src_SRC_FILES ${_SRC_FILES} PARENT_SCOPE)
set(src_INCLUDE_DIRS ${_INCLUDE_DIRS} PARENT_SCOPE)
